1. for login credentials and token

/api/login (post)
{
	"email" : "test_student@gmail.com"
}


2. for logout

/api/logout (get)

3. for getting student details

/api/studinfo/<int:student_id> (get)

4. for reporting issues

/api/report (post)
{
	"issue_type" : "content",
	"user_id" : 3,
	"course_id" : 2,
	"subject" : "API Test",
	"description" : "API Testing working"
}

5. for getting issues user reported

/api/report/<int:user_id> (get)

6. for getting tasks by user

/api/tasks/<int:user_id> (get)

7. for adding tasks by user

/api/tasks (post)
{
	"user_id" : "2",
	"title" : "API Test",
	"description" : "API testing deadline",
	"task_date" : "2025-04-13"
} 


8. for deleting tasks

/api/tasks/delete/<int:task_id> (delete)

9. to get course content

/api/content/<int:course_id> (get)

10. to post new course content

/api/content (post)
{
	"course_id" : 2,
        "content_name" : "Lecture 3.1",
        "content_type" : "Week 3",
        "transcript_url" : "https://drive.google.com/file/d/1_h32G_GIQtG-6ZUdEUbWaXR7Dx8f9HRL/view?usp=drive_link",
	"url": "https://www.youtube.com/watch?v=nL-086Wy4GA"
}

11. to edit the content

/api/content/edit/<int:content_id> (put)
{

        "content_name" : "Lecture 2.1",
